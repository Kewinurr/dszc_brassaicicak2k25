1.5.A)
	SELECT szolgaltatasok.szolgaltat_nev
	FROM epuletek INNER JOIN szolgaltatasok ON epuletek.epulet_azon = szolgaltatasok.szolgaltat_epazon
	WHERE (((epuletek.[epulet_azon])=[PARAMETERS epulet_azon Short]));
1.5.B)
	SELECT projektek.projekt_nev, SUM(projektek.projekt_koltseg) AS OsszesKoltseg
	FROM epuletek, projektek
	WHERE epuletek.epulet_tipus = 'Iskola' AND epuletek.epulet_azon = projektek.projekt_epazon
	GROUP BY projektek.projekt_nev;
1.5.C)
	SELECT szolgaltatasok.szolgaltat_tipus, Count(epuletek.epulet_azon) AS CountOfepulet_azon
	FROM epuletek INNER JOIN szolgaltatasok ON epuletek.epulet_azon = szolgaltatasok.szolgaltat_epazon
	GROUP BY szolgaltatasok.szolgaltat_tipus;
1.5.D)
	SELECT COUNT(*) AS NincsMeg
	FROM epuletek
	WHERE epuletek.epulet_azon NOT IN (SELECT szolgaltatasok.szolgaltat_epazon FROM szolgaltatasok);
1.5.E)
	SELECT epuletek.epulet_nev
	FROM epuletek INNER JOIN szolgaltatasok ON epuletek.epulet_azon = szolgaltatasok.szolgaltat_epazon
	GROUP BY epuletek.epulet_nev
	HAVING COUNT(szolgaltatasok.szolgaltat_epazon) > 2;
